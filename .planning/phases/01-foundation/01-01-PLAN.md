    1. `xpForLevel(level: number, xp: { baseXp: number; exponent: number; linearXp: number; multiplier: number }): number`
       - Formula: `Math.floor((xp.baseXp * Math.pow(level, xp.exponent) + xp.linearXp * level) * xp.multiplier)`
       - Comment: "XP required to reach a given level. Formula inferred from XpCurveConfig field semantics -- exact Kotlin computation not in reference/."

    2. `mobHpAtLevel(tier: { baseHp: number; hpPerLevel: number }, level: number): number`
       - Formula: `tier.baseHp + tier.hpPerLevel * level`

    3. `mobAvgDamageAtLevel(tier: { baseMinDamage: number; baseMaxDamage: number; damagePerLevel: number }, level: number): number`
       - Formula: `(tier.baseMinDamage + tier.baseMaxDamage) / 2 + tier.damagePerLevel * level`

    4. `mobAvgGoldAtLevel(tier: { baseGoldMin: number; baseGoldMax: number; goldPerLevel: number }, level: number): number`
       - Formula: `(tier.baseGoldMin + tier.baseGoldMax) / 2 + tier.goldPerLevel * level`

    5. `statBonus(statValue: number, divisor: number): number`
       - Formula: `Math.floor(statValue / divisor)`

    6. `dodgeChance(dodgeStatValue: number, bindings: { dodgePerPoint: number; maxDodgePercent: number }): number`
       - Formula: `Math.min(bindings.dodgePerPoint * dodgeStatValue, bindings.maxDodgePercent)`

    7. `playerHpAtLevel(level: number, rewards: { baseHp: number; hpPerLevel: number }, classHpPerLevel: number, hpScalingStat: number, hpScalingDivisor: number): number`
       - Formula: `rewards.baseHp + (rewards.hpPerLevel + classHpPerLevel + statBonus(hpScalingStat, hpScalingDivisor)) * level`
       - Comment: "Approximation -- exact server formula involves full player model. Stat bonus scales per level."

    8. `regenIntervalMs(statValue: number, regen: { baseIntervalMillis: number; minIntervalMillis: number }, hpRegenMsPerPoint: number): number`
       - Formula: `Math.max(regen.baseIntervalMillis - hpRegenMsPerPoint * statValue, regen.minIntervalMillis)`
