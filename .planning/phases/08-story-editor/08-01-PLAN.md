function snapshotStory(state: StoryState): Pick<StoryState, "storyPast" | "storyFuture"> {
  const past = [...state.storyPast, structuredClone(state.stories)];
  if (past.length > MAX_STORY_HISTORY) past.shift();
  return { storyPast: past, storyFuture: [] };
}

interface StoryState {
  stories: Record<string, Story>;
  dirty: Record<string, boolean>;
  activeStoryId: string | null;

}

    Specific implementation details:

    **addScene:** Append scene to `existing.scenes` with `sortOrder = existing.scenes.length`. If `existing.scenes.length === 0` (first scene), also set `activeSceneId: scene.id`.

    **removeScene:** Filter out scene by ID, then `.map((sc, i) => ({ ...sc, sortOrder: i }))` to normalize. Auto-selection logic: if `s.activeSceneId === sceneId`, select the scene at the same index (clamped to new length - 1), or null if no scenes remain.

    **reorderScenes:** Build a `Map<string, Scene>` from existing scenes. Map over the `sceneIds` array, look up each from the map, filter out any not found, then `.map((sc, i) => ({ ...sc!, sortOrder: i }))`.

    **updateScene:** Map over `existing.scenes`, replacing the matching scene with `{ ...sc, ...patch }`. No-op if scene ID not found in the story.

    **duplicateScene:** Find the original scene by ID. Clone it with `{ ...original, id: generateSceneId(), sortOrder: 0 }` (sortOrder will be renumbered). Splice the clone into scenes array after the original's index. Renumber all sortOrder values. Set `activeSceneId` to the new scene's ID.

    For `generateSceneId`, add a module-level function: `const generateSceneId = (): string => 'scene_' + Math.random().toString(36).substring(2, 8);`. Export it for use in UI components.

    export const selectActiveScene = (s: StoryState) => {

      const story = s.stories[s.activeStoryId];
      return story?.scenes.find((sc) => sc.id === s.activeSceneId);
    };
