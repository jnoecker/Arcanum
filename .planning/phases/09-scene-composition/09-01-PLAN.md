   ```typescript
   export const PRESET_SLOTS: Record<EntitySlot, { x: number; y: number }> = {
     "front-left":    { x: 20, y: 72 },
     "front-center":  { x: 50, y: 72 },
     "front-right":   { x: 80, y: 72 },
     "back-left":     { x: 25, y: 48 },
     "back-center":   { x: 50, y: 48 },
     "back-right":    { x: 75, y: 48 },
   };

   export const SLOT_ORDER: EntitySlot[] = [
     "front-center", "front-left", "front-right",
     "back-center", "back-left", "back-right",
   ];

   export function getNextSlot(occupiedSlots: EntitySlot[]): EntitySlot
   export function resolveEntityPosition(entity: SceneEntity): { x: number; y: number }
   export function isBackRow(slot?: EntitySlot): boolean
   export function getEntityScale(entity: SceneEntity): number  // 1.0 front, 0.78 back
   export function clampPosition(pos: { x: number; y: number }): { x: number; y: number }
   export function extractPlainText(narrationJson: string): string
   ```

   Implementation details:
   - `getNextSlot`: Find first slot in SLOT_ORDER not in occupiedSlots. Fallback to "front-center".
   - `resolveEntityPosition`: If entity.position, return it. If entity.slot, return PRESET_SLOTS[slot]. Fallback PRESET_SLOTS["front-center"].
   - `isBackRow`: Returns true if slot starts with "back-". False for undefined or front-* slots.
   - `getEntityScale`: If entity.slot and isBackRow(entity.slot), return 0.78. Else 1.0.
   - `clampPosition`: Return { x: Math.max(0, Math.min(100, pos.x)), y: same for y }.
   - `extractPlainText`: Parse JSON, walk content nodes recursively. Concatenate text nodes. Join paragraphs with "\n". Return "" on parse error or empty input.
