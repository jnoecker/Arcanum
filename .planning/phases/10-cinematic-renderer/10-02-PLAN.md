    ```typescript
    const handleUpdateEntity = useCallback((entityId: string, patch: Partial<SceneEntity>) => {
      const entities = scene.entities ?? [];
      const updated = entities.map((ent) =>
        ent.id === entityId ? { ...ent, ...patch } : ent,
      );
      updateScene(storyId, scene.id, { entities: updated });
    }, [scene.entities, scene.id, storyId, updateScene]);
    ```

    ```typescript
    const handleNarrationSpeedChange = useCallback((speed: NarrationSpeed | undefined) => {
    ```
