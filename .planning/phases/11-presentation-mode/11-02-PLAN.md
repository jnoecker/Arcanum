**F5 keyboard shortcut (add as a new useEffect after the existing auto-select scene effect around line 127):**
```typescript
useEffect(() => {
  if (!story) return;
  function handleF5(e: KeyboardEvent) {
    if (e.key === "F5" && canPresent && !isPresenting) {
      e.preventDefault();
      setIsPresenting(true);
    }
  }
  window.addEventListener("keydown", handleF5);
  return () => window.removeEventListener("keydown", handleF5);
}, [story, canPresent, isPresenting]);
```

**Exit handler (add as a useCallback after the handleCoverImageSelect callback around line 153):**
```typescript
const handlePresentationExit = useCallback(

    setIsPresenting(false);
    setActiveScene(exitSceneId);
  },
  [setActiveScene],
);
```
